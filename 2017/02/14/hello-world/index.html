int lowbit(int x){return x&-x;}
int sum(int* s,int t){
    int now=t,sum=0;
    while(now>0){
        sum+=s[now];
        now-=lowbit(now);
    }
    return sum;
}
void add(int* s,int siz,int t,int d){
    int now=t;

        now+=lowbit(now);
